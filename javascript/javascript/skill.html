# 格式化金钱

const ThousandNum = (num) =>
  num.toString().replace(/\B(?=(\d{3})+(?!\d))/g, ",")
const money = ThousandNum(20190214) // money => "20,190,214"

# 生成随机 ID

const RandomId = (len) =>
  Math.random()
    .toString(36)
    .substr(3, len)
const id = RandomId(10) // id => "jg7zpgiqva"

# 生成星级评分

const StartScore = (rate) => "★★★★★☆☆☆☆☆".slice(5 - rate, 10 - rate)
const start = StartScore(3) // start => "★★★"

# 补零

const FillZero = (num, len) => num.toString().padStart(len, "0")
const num = FillZero(169, 5) // num => "00169"

# 时间戳

const timestamp = +newDate("2019-02-14") // timestamp => 1550102400000

# 清空数组

const arr = [0, 1, 2]
arr.length = 0 // arr => []

# 克隆对象

const _obj = { a: 0, b: 1, c: 2 } // 以下方法任选一种
const obj = { ..._obj }
const obj = JSON.parse(JSON.stringify(_obj)) // obj => { a: 0, b: 1, c: 2 }

# 合并对象

const obj1 = { a: 0, b: 1, c: 2 }
const obj2 = { c: 3, d: 4, e: 5 }
const obj = { ...obj1, ...obj2 } // obj => { a: 0, b: 1, c: 3, d: 4, e: 5 }
